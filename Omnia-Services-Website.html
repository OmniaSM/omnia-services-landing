                            }

                        } catch (error) {

                            console.error("Firebase authentication error:", error);

                        }

                    }

                    

                    // After auth is resolved, set the user ID and initialize state

                    if (auth.currentUser) {

                        userId = auth.currentUser.uid;

                    } else {

                        // If all auth attempts fail (rare), use a temp ID

                        userId = 'anon-' + crypto.randomUUID();

                    }

                    

                    dbReady = true;

                    leadsCollectionPath = `/artifacts/${appId}/users/${userId}/leads`;

                    console.log("Firebase initialized for Form Submission. User ID:", userId);

                });


            } catch (e) {

                console.error("Firebase Initialization Failed:", e);

            }
